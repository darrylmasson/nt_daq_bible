

\newglossaryentry{ZLE}{
  name=ZLE,
  description={Zero-Length Encoding. A digitizer feature applied to (usually) a longer waveform, where regions of zeros (samples below threshold) are suppressed, and encoded into the waveform via control words indicating the number of suppressed samples.}
}

\newglossaryentry{HEV}{
  name=HEV,
  description={High-energy veto. Some fancy logic coded into a Skutek DDC-10 that allows us to veto digitizers if we see high-energy events. Primarily a technique to reduce storage for data that no one cares about (high-energy events in calibration data), also keeps the effective livetime up as we don't worry about going busy during electron trains.}
}
